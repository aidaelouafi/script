# ===============================
# ⚡ CACHE INTELLIGENT
# ===============================
CACHE = {}
CACHE_TIME = {}
TTL = 300  # 5 minutes

def run_hive(query):
    now = time.time()

    if query in CACHE and now - CACHE_TIME[query] < TTL:
        return CACHE[query]

    cmd = f'hive -S -e "USE default; {query}"'
    result = subprocess.run(cmd, shell=True, capture_output=True, text=True)

    lignes = [
        l.split("\t")
        for l in result.stdout.strip().split("\n")
        if l and not l.startswith("WARNING")
    ]

    CACHE[query] = lignes
    CACHE_TIME[query] = now

    return lignes

def query_A():
    q = "SELECT product_id AS TOP_10_produits_vendus FROM ( SELECT product_id, COUNT(*) AS nbr_produit_vendu FROM order_items GROUP BY product_id ) sub ORDER BY product_id DESC LIMIT 10"
    return run_hive(q), ["TOP_10_produits_vendus"]

def query_B():
    q = "SELECT product_id, COUNT(*) AS nombre_de_produit_vendu FROM order_items GROUP BY product_id ORDER BY nombre_de_produit_vendu DESC"
    return run_hive(q), ["product_id", "nombre_de_produit_vendu"]

def query_C():
    q = "SELECT SUM(price + freight_value) AS chiffre_affaires_total FROM order_items"
    return run_hive(q), ["chiffre_affaires_total"]

def query_D():
    q = "SELECT o.customer_id, SUM(oi.price + oi.freight_value) AS total_depense FROM orders o JOIN order_items oi ON o.order_id = oi.order_id GROUP BY o.customer_id ORDER BY total_depense DESC LIMIT 10"
    return run_hive(q), ["customer_id", "total_depense"]

def query_E_qty():
    q = "SELECT product_id, COUNT(*) AS quantite_vendue FROM order_items GROUP BY product_id ORDER BY quantite_vendue DESC LIMIT 10"
    return run_hive(q), ["product_id", "quantite_vendue"]

def query_E_ca():
    q = "SELECT product_id, SUM(price) AS ca_produit FROM order_items GROUP BY product_id ORDER BY ca_produit DESC LIMIT 10"
    return run_hive(q), ["product_id", "ca_produit"]

def query_F():
    q = "SELECT AVG(DATEDIFF(order_delivered_customer_date, order_estimated_delivery_date)) AS retard_moyen FROM orders WHERE order_delivered_customer_date IS NOT NULL"
    return run_hive(q), ["retard_moyen"]

def query_G():
    q = "SELECT AVG(DATEDIFF(order_delivered_customer_date, order_approved_at)) AS delai_moyen_approbation_livraison FROM orders WHERE order_delivered_customer_date IS NOT NULL"
    return run_hive(q), ["delai_moyen_approbation_livraison"]

def query_H():
    q = "SELECT customer_id, COUNT(*) AS nb_commandes FROM orders GROUP BY customer_id ORDER BY nb_commandes DESC LIMIT 10"
    return run_hive(q), ["customer_id", "nb_commandes"]

def query_I():
    q = "SELECT ROUND(100 * SUM(CASE WHEN order_delivered_customer_date > order_estimated_delivery_date THEN 1 ELSE 0 END) / COUNT(*), 2) AS pourcentage_retard FROM orders WHERE order_delivered_customer_date IS NOT NULL"
    return run_hive(q), ["pourcentage_retard"]

def query_J():
    q = "SELECT customer_id, COUNT(DISTINCT DATE(order_purchase_timestamp)) AS nb_jours_commande FROM orders GROUP BY customer_id HAVING nb_jours_commande > 1 ORDER BY nb_jours_commande DESC"
    return run_hive(q), ["customer_id", "nb_jours_commande"]

def query_K():
    q = "SELECT o.customer_id, AVG(oi.price + oi.freight_value) AS panier_moyen FROM orders o JOIN order_items oi ON o.order_id = oi.order_id GROUP BY o.customer_id ORDER BY panier_moyen DESC"
    return run_hive(q), ["customer_id", "panier_moyen"]

def query_L():
    q = "SELECT product_id, AVG(price - freight_value) AS marge_moyenne FROM order_items GROUP BY product_id ORDER BY marge_moyenne ASC LIMIT 10"
    return run_hive(q), ["product_id", "marge_moyenne"]

def query_M():
    q = "SELECT product_id, COUNT(*) AS nb_ventes FROM order_items GROUP BY product_id ORDER BY nb_ventes ASC LIMIT 10"
    return run_hive(q), ["product_id", "nb_ventes"]

def query_N1():
    q = "SELECT order_status, COUNT(*) AS nb_commandes FROM orders WHERE order_status IN ('canceled', 'returned') GROUP BY order_status"
    return run_hive(q), ["order_status", "nb_commandes"]

def query_N2():
    q = "SELECT order_id, DATEDIFF(order_delivered_customer_date, order_purchase_timestamp) AS delai_total FROM orders ORDER BY delai_total DESC LIMIT 10"
    return run_hive(q), ["order_id", "delai_total"]
