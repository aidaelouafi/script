from flask import Flask, render_template_string
import subprocess

app = Flask(__name__)

# Initialisation Hive
def init_hive():
    commandes = """
    CREATE DATABASE IF NOT EXISTS entreprise;
    USE entreprise;

    CREATE TABLE IF NOT EXISTS employe_ext (
        id INT,
        nom STRING,
        age INT,
        salaire INT,
        departement STRING
    );

    INSERT INTO employe_ext VALUES
    (1,'Mohamed',30,20000,'RH'),
    (2,'Manal',24,15000,'BIGDATA'),
    (3,'Aida',25,16000,'RH'),
    (4,'Youssef',28,18000,'IT'),
    (5,'Sara',27,17000,'FINANCE');
    """
    subprocess.run(f'hive -e "{commandes}"', shell=True)

# Exécuter requête
def run_query(query):
    cmd = f'hive -S -e "USE entreprise; {query}"'
    result = subprocess.run(cmd, shell=True, capture_output=True, text=True)
    return result.stdout

# Interface HTML
html = """
<h2>Dashboard Hive</h2>

<button onclick="window.location.href='/all'">Tous les employés</button>
<button onclick="window.location.href='/top'">Top salaires</button>
<button onclick="window.location.href='/avg'">Moyenne salaire</button>

<pre>{{result}}</pre>
"""

@app.route("/")
def home():
    return render_template_string(html, result="Clique sur un bouton")

@app.route("/all")
def all_data():
    return render_template_string(html, result=run_query("SELECT * FROM employe_ext"))

@app.route("/top")
def top():
    return render_template_string(html, result=run_query("SELECT nom, salaire FROM employe_ext ORDER BY salaire DESC LIMIT 5"))

@app.route("/avg")
def avg():
    return render_template_string(html, result=run_query("SELECT departement, AVG(salaire) FROM employe_ext GROUP BY departement"))

if __name__ == "__main__":
    init_hive()
    app.run(host="0.0.0.0", port=5000, debug=True)
